{\centering{\Huge\ding{96}\quad{\bsifamily Ein kleines Märchen}\quad\ding{96}} \par\bigskip

{\raggedright{\bsifamily Unsere kleine Geschichte beginnt auf einer wunderschönen Blumenwiese im Land des Königs Schwabacher. Dort traf der König eines Tages auf die noch viel wunderschönere Prinzessin}\textsc{von kapitälchen.}\\} \par\bigskip



\begin{verse}

\textsc{von kapitälchen:} \textit{{\Large OHA!} Wer sind sie denn? Sie haben mich aber erschreckt.}\\

\textsc{schwabacher:} {\bfseries Ich Bin der König Schwabacher. Mit wem habe ichdie Ehre?}\\

\textsc{von kapitälchen:} \textit{Mein Name ist Prinzessin \textsc{Von Kapitälchen}. Sehr erfreut sie kennenzulernen! Was machen Sie an einem solch wunderschönen Tag, auf dieser wunderschönen Blumenwiese?}\\

\textsc{schwabacher:} {\bfseries Tut mir leid meine Liebe. Ich bin leider ein wenig schwerhörig\footnote{Durch einen tragischen Kanonenunfall}. Können Sie nicht ein bisschen lauter sprechen?}\\

\textsc{von kapitälchen:} \textit{{\LARGE Was machen Sie an einem solch wunderschönen Tag, auf dieser wunderschönen Blumenwiese?}}\\

\textsc{schwabacher:} {\bfseries Verzeihung? Wieso passt der Sarg nicht für den Riesen?}\\

\textsc{von kapitälchen:} \textit{{\huge Ich habe gefragt, was Sie an einem
solch wunderschönen Tag auf dieser wunderschönen Blumenwiese machen.}}\\

\textsc{schwabacher:} {\bfseries {\huge HÄH!}}\\

\textsc{von kapitälchen:} \textit{{\huge Ich habe gefragt...}{\Huge Ach geh doch dahin wo der Pfeffer wächst!}}\par\bigskip
\end{verse}
